## 测试环境准备

### 1. 启动 PostgreSQL 数据库

使用 Docker 启动测试数据库:

```bash
docker run -d \
  --name metering-postgres-test \
  -e POSTGRES_USER=metering_user \
  -e POSTGRES_PASSWORD=test_password \
  -e POSTGRES_DB=metering_test \
  -p 5432:5432 \
  postgres:14-alpine

# 等待数据库启动
sleep 5

# 验证数据库可连接
docker exec metering-postgres-test psql -U metering_user -d metering_test -c "SELECT version();"
```

### 2. 设置环境变量

```bash
export DB_HOST=localhost
export DB_PORT=5432
export DB_NAME=metering_test
export DB_USER=metering_user
export DB_PASSWORD=test_password
```

### 测试 3: 用户操作测试

**测试目标**: 验证用户 CRUD 操作是否正常

**测试步骤**:

编写测试脚本 `test_user_operations.go`:

```go
package main

import (
    "log"
    "github.com/openmesh/metering-service/internal/database"
)

func main() {
    // 连接数据库
    db, err := database.NewDatabase("host=localhost port=5432 user=metering_user password=test_password dbname=metering_test sslmode=disable")
    if err != nil {
        log.Fatalf("Failed to connect: %v", err)
    }
    defer db.Close()

    // 测试 1: 创建用户
    err = db.CreateUser("test_user_001", 1048576) // 1MB
    if err != nil {
        log.Fatalf("CreateUser failed: %v", err)
    }
    log.Println("✓ CreateUser success")

    // 测试 2: 获取用户
    user, err := db.GetUser("test_user_001")
    if err != nil {
        log.Fatalf("GetUser failed: %v", err)
    }
    if user.Quota != 1048576 || user.Used != 0 || user.Status != "active" {
        log.Fatalf("User data incorrect: %+v", user)
    }
    log.Println("✓ GetUser success")

    // 测试 3: 增加流量 (原子性测试)
    err = db.IncrementTraffic("test_user_001", 1024, 2048)
    if err != nil {
        log.Fatalf("IncrementTraffic failed: %v", err)
    }
    user, _ = db.GetUser("test_user_001")
    if user.Used != 3072 {
        log.Fatalf("Traffic increment incorrect: expected 3072, got %d", user.Used)
    }
    log.Println("✓ IncrementTraffic success")

    // 测试 4: 更新用户状态
    err = db.UpdateUserStatus("test_user_001", "blocked")
    if err != nil {
        log.Fatalf("UpdateUserStatus failed: %v", err)
    }
    user, _ = db.GetUser("test_user_001")
    if user.Status != "blocked" {
        log.Fatalf("Status update incorrect: %s", user.Status)
    }
    log.Println("✓ UpdateUserStatus success")

    // 测试 5: 更新配额
    err = db.UpdateUserQuota("test_user_001", 2097152) // 2MB
    if err != nil {
        log.Fatalf("UpdateUserQuota failed: %v", err)
    }
    user, _ = db.GetUser("test_user_001")
    if user.Quota != 2097152 {
        log.Fatalf("Quota update incorrect: %d", user.Quota)
    }
    log.Println("✓ UpdateUserQuota success")

    // 测试 6: 获取所有用户
    users, err := db.GetAllUsers()
    if err != nil {
        log.Fatalf("GetAllUsers failed: %v", err)
    }
    if len(users) == 0 {
        log.Fatalf("GetAllUsers returned empty")
    }
    log.Println("✓ GetAllUsers success")

    log.Println("\n✅ All user operations passed!")
}
```

运行测试:

**预期结果**:

```go
package main

import (
    "log"
    "time"
    "github.com/openmesh/metering-service/internal/database"
)

func main() {
    db, err := database.NewDatabase("host=localhost port=5432 user=metering_user password=test_password dbname=metering_test sslmode=disable")
    if err != nil {
        log.Fatalf("Failed to connect: %v", err)
```

```go
    start := time.Now()
    _, err = db.GetUser("perf_user_050")

        log.Fatalf("Query too slow: %v (expected < 50ms)", duration)
    }
    log.Printf("✓ Single query: %v (< 50ms)", duration)

    }
    duration = time.Since(start)
    if duration > 1*time.Second {
        log.Fatalf("Batch insert too slow: %v (expected < 1s)", duration)
    }
    log.Printf("✓ Batch insert 1000 logs: %v (< 1s)", duration)

    // 测试 3: 流量增量性能
    start = time.Now()
    err = db.IncrementTraffic("perf_user_001", 1024, 2048)
    duration = time.Since(start)
    if duration > 10*time.Millisecond {
        log.Fatalf("Increment too slow: %v (expected < 10ms)", duration)
    }
    log.Printf("✓ Traffic increment: %v (< 10ms)", duration)

    log.Println("\n✅ All performance tests passed!")
```

### 测试 7: 并发测试

**测试目标**: 验证数据库在并发场景下的正确性

**测试步骤**:

编写并发测试脚本 `test_concurrency.go`:

```go
package main

import (
    "log"
    "sync"
    "github.com/openmesh/metering-service/internal/database"
)

func main() {
    db, err := database.NewDatabase("host=localhost port=5432 user=metering_user password=test_password dbname=metering_test sslmode=disable")
    if err != nil {
        log.Fatalf("Failed to connect: %v", err)
    }
    defer db.Close()

    // 准备测试用户
    db.CreateUser("concurrent_user", 0)

    // 并发增加流量
    var wg sync.WaitGroup
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for j := 0; j < 100; j++ {
                db.IncrementTraffic("concurrent_user", 1024, 0)
            }
        }()
    }
    wg.Wait()

    // 验证结果
    user, _ := db.GetUser("concurrent_user")
    expected := int64(10 * 100 * 1024) // 10 goroutines * 100 iterations * 1024 bytes
    if user.Used != expected {
        log.Fatalf("Concurrent increment incorrect: expected %d, got %d", expected, user.Used)
    }
    log.Printf("✓ Concurrent increment correct: %d", user.Used)

    log.Println("\n✅ Concurrency test passed!")
}
```

**预期结果**:
- [ ] 并发操作无数据竞争
- [ ] 流量统计准确,无丢失
